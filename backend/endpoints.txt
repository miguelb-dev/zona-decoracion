json
{
  "idPinturaInventario": int (1 o 2),
  "proveedor": "String",
  "fechaRegistro": "2026-06-24",
  "cantidadRegistrada": 100.00,
  "registradoPor": "String",
  "observaciones": "String"
}

json
{
  "mensaje": "Registro procesado con éxito",
  "id": int (1 o 2)
}
